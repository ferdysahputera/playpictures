Cara Setup Spreadsheet

1
Buat Google Spreadsheet baru di Google Drive.
2
Klik menu Extensions → Apps Script. Hapus semua kode default, lalu paste kode di bawah ini:
function doGet(e){ try{ var d=read(); d._ts=new Date().getTime(); return ContentService .createTextOutput(JSON.stringify(d)) .setMimeType(ContentService.MimeType.JSON); }catch(err){ return ContentService .createTextOutput(JSON.stringify({error:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e){ try{ var d=JSON.parse(e.postData.contents); write(d); d._ts=new Date().getTime(); return ContentService .createTextOutput(JSON.stringify({status:'ok',ts:d._ts})) .setMimeType(ContentService.MimeType.JSON); }catch(err){ return ContentService .createTextOutput(JSON.stringify({error:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function read(){ var ss=SpreadsheetApp.getActiveSpreadsheet(); var s=ss.getSheetByName('Settings'); if(!s){s=ss.insertSheet('Settings');s.getRange('A1').setValue('{}');} var v=s.getRange('A1').getValue(); return v?JSON.parse(v):{}; } function write(d){ var ss=SpreadsheetApp.getActiveSpreadsheet(); var s=ss.getSheetByName('Settings'); if(!s){s=ss.insertSheet('Settings');} s.getRange('A1').setValue(JSON.stringify(d,null,2)); }
3
Klik Deploy → New deployment. Isi:
Description: Mosque API
Execute as: Me
Who has access: Anyone
4
Salin Web app URL yang muncul, lalu tempel di kolom URL di bawah. Format: https://script.google.com/macros/s/AKfycbyoV4VImRK-YnHo2wLS65lT6Lk56cVgVlKZSH6AZDpe0lZs7oQ0I47UvmsQ7GPIKhoyMg/exec
5
Klik "Test Koneksi". Jika hijau, semuanya sudah benar.
