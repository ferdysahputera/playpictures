Koneksi Google Spreadsheet

Masukkan Spreadsheet ID dari URL Google Sheets. Pastikan Apps Script sudah terpasang (lihat panduan).

Panduan Apps Script
1. Buka Google Spreadsheet Anda
2. Klik Extensions > Apps Script
3. Hapus kode default, paste kode di bawah:
4. Klik Deploy > New Deployment
5. Pilih type: Web app
6. Execute as: Me
7. Who has access: Anyone
8. Klik Deploy, copy URL-nya

// ========= CODE APPS SCRIPT =========
function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName('Config');
  if (!sheet) return ContentService.createTextOutput('Sheet not found');
  var data = sheet.getDataRange().getValues();
  var result = [];
  for (var i = 1; i < data.length; i++) {
    if (data[i][0]) result.push({key: data[i][0], value: data[i][1]});
  }
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  var params = JSON.parse(e.postData.contents);
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName('Config');
  if (!sheet) return ContentService.createTextOutput('Sheet not found');
  for (var i = 0; i < params.length; i++) {
    var row = findRow(sheet, params[i].key);
    if (row > 0) {
      sheet.getRange(row, 2).setValue(params[i].value);
    } else {
      sheet.appendRow([params[i].key, params[i].value]);
    }
  }
  return ContentService.createTextOutput('OK');
}

function findRow(sheet, key) {
  var data = sheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] == key) return i + 1;
  }
  return 0;
}
