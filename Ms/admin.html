Panduan Apps Script
1. Buka Spreadsheet > Extensions > Apps Script
2. Hapus kode default, paste:
function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(e.parameter.sheet || 'Config');
  if (!sheet) return ContentService.createTextOutput(JSON.stringify({error:'Not found'})).setMimeType(ContentService.MimeType.JSON);
  var data = sheet.getDataRange().getValues(), result = [];
  for (var i = 1; i < data.length; i++) { if (data[i][0]) result.push({key:String(data[i][0]).trim(), value:data[i][1]!==undefined?data[i][1]:''}); }
  return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var params = JSON.parse(e.postData.contents);
    if (!Array.isArray(params)) params = [params];
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName(params[0].sheet || 'Config');
    if (!sheet) return ContentService.createTextOutput(JSON.stringify({error:'Not found'})).setMimeType(ContentService.MimeType.JSON);
    if (sheet.getLastRow() === 0) sheet.appendRow(['Key', 'Value']);
    var updated = 0, added = 0;
    for (var i = 0; i < params.length; i++) {
      if (!params[i].key || params[i].key === 'sheet') continue;
      var row = findRow(sheet, params[i].key);
      if (row > 0) { sheet.getRange(row, 2).setValue(params[i].value); updated++; }
      else { sheet.appendRow([params[i].key, params[i].value]); added++; }
    }
    return ContentService.createTextOutput(JSON.stringify({status:'OK', updated:updated, added:added})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) { return ContentService.createTextOutput(JSON.stringify({error:err.message})).setMimeType(ContentService.MimeType.JSON); }
}

function findRow(sheet, key) {
  var data = sheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) { if (String(data[i][0]).trim() === String(key).trim()) return i + 1; }
  return 0;
}
